 /* Node is defined as :
 class Node 
    int data;
    Node left;
    Node right;
 */
// https://www.hackerrank.com/challenges/binary-search-tree-insertion
static Node Insert(Node root, int value)
{
    if (root == null) {
        root = new Node();
        root.data = value;
        
        return root;
    }
    
    if (value < root.data) 
    { 
        root.left = Insert(root.left, value);
    } 
    else if (value > root.data) 
    {
        root.right = Insert(root.right, value);
    }
    
    return root;
}
